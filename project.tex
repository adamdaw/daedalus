\usepackage{fancyvrb}
\usepackage[nobottomtitles*]{titlesec}
\usepackage{graphicx}
\usepackage{fancyhdr}
\usepackage{etoolbox}

%%─── Typographic safety ─────────────────────────────────────────────────────
%% nobottomtitles* (above): pushes headings that would land in the bottom 20%
%% of the text area to the next page — prevents widow headings.
%% emergencystretch: allows TeX to add up to 2em of inter-word space as a last
%% resort, preventing inline code and long words from overflowing into margins.
\emergencystretch=2em

\makeatletter

%%─── Subtitle support ────────────────────────────────────────────────────────
%% Pandoc calls \subtitle{...} from config.yaml; article class doesn't define
%% it, so we define it here and store the value for use in \maketitle.
\providecommand{\subtitle}[1]{\gdef\@subtitle{#1}}
\gdef\@subtitle{}

%%─── Additional cover page metadata ─────────────────────────────────────────
%% Override any of these in config.yaml via header-includes, e.g.:
%%   header-includes:
%%     - \def\docclient{Acme Corp}
%%     - \def\docversion{1.0}
%%     - \def\docclassification{Confidential}
\providecommand{\docclient}{}
\providecommand{\docversion}{}
\providecommand{\docclassification}{}

%%─── Running headers and footers ─────────────────────────────────────────────
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\itshape\@title}
\fancyhead[R]{\small\@author}
\fancyfoot[C]{\small\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

%%─── Page break before each top-level section ────────────────────────────────
\newcommand{\sectionbreak}{\clearpage}

%%─── Cover page logo ─────────────────────────────────────────────────────────
%% Checks both the direct filename (found when TEXINPUTS includes the images
%% directory) and the images/ subdirectory path (found when xelatex runs from
%% the repository root). Supports .jpg, .png, and .pdf logo formats.
\newcommand{\insertlogo}{%
  \IfFileExists{logo.jpg}{\includegraphics[width=0.35\textwidth]{logo.jpg}\\[2cm]}{%
  \IfFileExists{logo.png}{\includegraphics[width=0.35\textwidth]{logo.png}\\[2cm]}{%
  \IfFileExists{logo.pdf}{\includegraphics[width=0.35\textwidth]{logo.pdf}\\[2cm]}{%
  \IfFileExists{images/logo.jpg}{\includegraphics[width=0.35\textwidth]{images/logo.jpg}\\[2cm]}{%
  \IfFileExists{images/logo.png}{\includegraphics[width=0.35\textwidth]{images/logo.png}\\[2cm]}{%
  \IfFileExists{images/logo.pdf}{\includegraphics[width=0.35\textwidth]{images/logo.pdf}\\[2cm]}{%
  \vspace{3.5cm}}}}}}}

%%─── Custom cover page ───────────────────────────────────────────────────────
\renewcommand{\maketitle}{%
  \begin{titlepage}%
    \centering%
    \vspace*{1.5cm}%
    \insertlogo%
    {\Huge\bfseries\@title\par}%
    \ifdefempty{\@subtitle}{}{%
      \vspace{0.5cm}{\Large\@subtitle\par}%
    }%
    \vspace{1.5cm}%
    {\large\@author\par}%
    \vspace{0.5cm}%
    {\large\@date\par}%
    \vspace{1cm}%
    \ifdefempty{\docclient}{}{%
      {\normalsize Prepared for: \docclient\par}%
    }%
    \ifdefempty{\docversion}{}{%
      \vspace{0.25cm}{\normalsize Version \docversion\par}%
    }%
    \vfill%
    \ifdefempty{\docclassification}{}{%
      {\small\itshape\docclassification\par}%
    }%
  \end{titlepage}%
}

\makeatother
